\section{Exercise 937a}

Given:
\[
y = x\ln{x}
\]

Solution:
\[
y' = 1\cdot\ln{x} + x\frac{1}{x} = \ln{x} + 1
\]

Result:
\[
y' = \ln{x} + 1
\]

\section{Exercise 937b}

Given:
\[
y = \frac{1 + \ln{x}}{x}
\]

Solution:
\[
y' = \frac{(0 + \frac{1}{x})x - (1 + \ln{x})\cdot1}{x^{2}} = \frac{1 - 1 - \ln{x}}{x^{2}} = \frac{-\ln{x}}{x^{2}} = -\frac{\ln{x}}{x^{2}}
\]

Result:
\[
y' = -\frac{\ln{x}}{x^{2}}
\]
