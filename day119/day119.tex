\section{Exercise 1050a}

Given:
\[
x^{2} + y^{2} = a^{2}
\]

Solution:
\[
2x + 2yy' = 0
\]
\[
x + yy' = 0
\]
\[
yy' = -x
\]
\[
y' = -\frac{x}{y}
\]

\[
y'' = -\frac{y - xy'}{y^{2}}
\]
\[
y'' = -\frac{y - x(-\frac{x}{y})}{y^{2}}
\]
\[
y'' = -\frac{y + \frac{x^{2}}{y}}{y^{2}}
\]
\[
y'' = -\frac{\frac{y^{2} + x^{2}}{y}}{y^{2}}
\]
\[
y'' = -\frac{y^{2} + x^{2}}{y^{3}}
\]

\[
x^{2} + y^{2} = a^{2}
\]
\[
y^{2} = a^{2} - x^{2}
\]

\[
y'' = -\frac{a^{2} - x^{2} + x^{2}}{y^{3}}
\]
\[
y'' = -\frac{a^{2}}{y^{3}}
\]

Result:
\[
y'' = -\frac{a^{2}}{y^{3}}
\]
