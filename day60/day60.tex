\section{Exercise 606}

Given:
\[
\left|
    \begin{array}{ccc}
    m + a & m - a & a\\
    n + a & 2n - a & a\\
    a & -a & a
    \end{array}
\right|
\]

Solution:
\[
(m + a)(2n - a)a = (2mn - am + 2an - a^{2})a = 2amn - a^{2}m + 2a^{2}n - a^{3}
\]
\[
(m - a)aa = ma^{2} - a^{3}
\]
\[
a(n + a)(-a) = -a^{2}(n + a) = -a^{2}n - a^{3}
\]
\[
a(2n - a)a = a^{2}(2n - a) = 2na^{2} - a^{3}
\]
\[
-aa(m + a) = -a^{2}m - a^{3}
\]
\[
a(n + a)(m - a) = a(nm - na + am - a^{2}) = anm - na^{2} + a^{2}m - a^{3}
\]
\\
\\
\[
2amn - a^{2}m + 2a^{2}n - a^{3} +
\]
\[
ma^{2} - a^{3} +
\]
\[
-a^{2}n - a^{3} -
\]
\[
(2na^{2} - a^{3}) -
\]
\[
(-a^{2}m - a^{3}) -
\]
\[
(anm - na^{2} + a^{2}m - a^{3})
\]
\\
\\
\[
2amn - a^{2}m + 2a^{2}n - a^{3} +
\]
\[
ma^{2} - a^{3} +
\]
\[
-a^{2}n - a^{3} +
\]
\[
-2na^{2} + a^{3} +
\]
\[
a^{2}m + a^{3} +
\]
\[
-anm + na^{2} - a^{2}m + a^{3} = amn
\]

Result:
\[
\left|
    \begin{array}{ccc}
    m + a & m - a & a\\
    n + a & 2n - a & a\\
    a & -a & a
    \end{array}
\right| = amn
\]
