\section{Exercise 1045a}

Given:
\[
x^{2} + xy + y^{2} = 6
\]

Solution:
\[
2x + y + xy' + 2yy' = 0
\]
\[
2x + y + y'(x + 2y) = 0
\]
\[
y'(x + 2y) = -(2x + y)
\]
\[
y' = -\frac{2x + y}{x + 2y}
\]

Result:
\[
y' = -\frac{2x + y}{x + 2y}
\]

\section{Exercise 1045b}

Given:
\[
x^{2} + y^{2} - xy = 0
\]

Solution:
\[
2x + 2yy' - (y + xy') = 0
\]
\[
2x + 2yy' - y - xy' = 0
\]
\[
2x - y + 2yy' - xy' = 0
\]
\[
2x - y + y'(2y - x) = 0
\]
\[
y'(2y - x) = y - 2x
\]
\[
y' = \frac{y - 2x}{2y - x}
\]

Result:
\[
y' = \frac{y - 2x}{2y - x}
\]
