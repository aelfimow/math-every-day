\section{Exercise 1044a}

Given:
\[
x^{2} + y^{2} = a^{2}
\]

Solution:
\[
2x + 2yy' = 0
\]
\[
2yy' = -2x
\]
\[
yy' = -x
\]
\[
y' = -\frac{x}{y}
\]

Result:
\[
y' = -\frac{x}{y}
\]

\section{Exercise 1044b}

Given:
\[
y^{2} = 2px
\]

Solution:
\[
2yy' = 2p
\]
\[
yy' = p
\]
\[
y' = \frac{p}{y}
\]

Result:
\[
y' = \frac{p}{y}
\]

\section{Exercise 1044c}

Given:
\[
\frac{x^{2}}{a^{2}} - \frac{y^{2}}{b^{2}} = 1
\]

Solution:
\[
\frac{2x}{a^{2}} - \frac{2yy'}{b^{2}} = 0
\]
\[
-\frac{2yy'}{b^{2}} = -\frac{2x}{a^{2}}
\]
\[
\frac{yy'}{b^{2}} = \frac{x}{a^{2}}
\]
\[
yy' = \frac{xb^{2}}{a^{2}}
\]
\[
y' = \frac{xb^{2}}{ya^{2}}
\]

Result:
\[
y' = \frac{xb^{2}}{ya^{2}}
\]
