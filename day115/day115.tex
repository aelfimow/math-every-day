\section{Exercise 1044a}

Given:
\[
x^{2} + y^{2} = a^{2}
\]

Solution:
\[
2x + 2yy' = 0
\]
\[
2yy' = -2x
\]
\[
yy' = -x
\]
\[
y' = -\frac{x}{y}
\]

Result:
\[
y' = -\frac{x}{y}
\]

\section{Exercise 1044b}

Given:
\[
y^{2} = 2px
\]

Solution:
\[
2yy' = 2p
\]
\[
yy' = p
\]
\[
y' = \frac{p}{y}
\]

Result:
\[
y' = \frac{p}{y}
\]
