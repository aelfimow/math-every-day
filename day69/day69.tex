\section{Exercise 880a}

Given:
\[
y = \sin{2}{x}
\]

Solution:
\[
y' = 2\sin{2 - 1}{x}\cos{x} = 2\sin{x}\cos{x} = \sin{2x}
\]

Result:
\[
y' = \sin{2x}
\]

\section{Exercise 880b}

Given:
\[
y = \cos{2}{x}
\]

Solution:
\[
y' = 2\cos{2 - 1}{x}(-\sin{x}) = -2\sin{x}\cos{x} = -\sin{2x}
\]

Result:
\[
y' = -\sin{2x}
\]
