\section{Exercise 596}

Given:
\[
\left|
    \begin{array}{ccc}
        a & -a &  a\\
        a &  a & -a\\
        a & -a & -a
    \end{array}
\right|
\]

Solution:
\[
\left|
    \begin{array}{ccc}
        a & -a &  a\\
        a &  a & -a\\
        a & -a & -a
    \end{array}
\right| =
aa(-a) + (-a)(-a)a + aa(-a) - aaa - (-a)(-a)a - (-a)a(-a)
\]

\[
-aaa + aaa - aaa - aaa - aaa - aaa = -4aaa = -4a^{3}
\]

Result:
\[
\left|
    \begin{array}{ccc}
        a & -a &  a\\
        a &  a & -a\\
        a & -a & -a
    \end{array}
\right| = -4a^{3}
\]
